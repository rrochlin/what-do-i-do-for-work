import pandas as pd
import sqlalchemy as sa

df = pd.read_excel("sales.xlsx")
df = df.rename(columns={"Sales$": "sales_usd"})
df["sales_eur"] = df["sales_usd"] * 0.92

engine = sa.create_engine("postgresql+psycopg2://user:pass@db/sales")
with engine.begin() as conn:
    df.to_sql("sales", conn, if_exists="replace", index=False)
